ICloud {
	data_item get(user_details, folder, data_uri)
	data_uri put(user_details, folder, data)
	bool post(user_details, folder, old_data_uri, data)
	bool delete(user_details, folder, data_uri)
}

WhiteBoardCloudProxy : ICloud {
	Cloud cloudService;
	folder = 'whiteboard'

	WhiteBoardCloudProxy(ICloud service) {
		cloudService = service;
	}

	data_item get(user_details, data_uri) {
		return cloudService.get(user_details, folder, data_uri);
	}

	data_uri put(user_details, folder, data) {
		return cloudService.data_uri put(user_details, folder, data)
	}

	bool post(user_details, folder, old_data_uri, data) {
		return cloudService.post(user_details, folder, old_data_uri, data)
	}

	bool delete(user_details, folder, data_uri) {
		return cloudService.delete(user_details, folder, data_uri)
	}
}
